Overview

Schema update required

The notes, CV categories, and leads features need a one-time SQL migration. Open your Supabase SQL editor, paste this and run it:

ALTER TABLE enquiries ADD COLUMN IF NOT EXISTS notes TEXT;
ALTER TABLE cv_submissions ADD COLUMN IF NOT EXISTS notes TEXT;
ALTER TABLE cv_submissions ADD COLUMN IF NOT EXISTS cv_category TEXT;
CREATE TABLE IF NOT EXISTS cv_categories (id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT UNIQUE NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW());
ALTER TABLE cv_categories ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "auth manage cv_categories" ON cv_categories FOR ALL TO authenticated USING (true) WITH CHECK (true);
CREATE TABLE IF NOT EXISTS leads (id UUID PRIMARY KEY DEFAULT gen_random_uuid(), email TEXT UNIQUE NOT NULL, name TEXT, phone TEXT, type TEXT DEFAULT 'candidate', company TEXT, sector TEXT, notes TEXT, status TEXT DEFAULT 'new', source TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW());
ALTER TABLE leads ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "anon insert leads" ON leads FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY IF NOT EXISTS "auth full leads" ON leads FOR ALL TO authenticated USING (true) WITH CHECK (true);
